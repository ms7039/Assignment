CREATE OR REPLACE TABLE RETAIL_MARKETING_DB.CAMPAIGN.CAMPAIGN_SALES AS
SELECT
    ss.ss_ticket_number                          AS TICKET_NUMBER,
    ss.ss_customer_sk                            AS CUSTOMER_ID,
    c.c_first_name || ' ' || c.c_last_name        AS CUSTOMER_NAME,
    d.d_date                                     AS SALE_DATE,
    d.d_year                                     AS SALE_YEAR,
    d.d_month_seq                                AS SALE_MONTH,
    i.i_item_desc                                AS ITEM_DESCRIPTION,
    i.i_category                                 AS ITEM_CATEGORY,
    ss.ss_quantity                               AS QUANTITY_SOLD,
    ss.ss_sales_price                            AS SALE_PRICE,
    ss.ss_net_paid                               AS NET_PAID,
    ss.ss_net_profit                             AS NET_PROFIT,
    ss.ss_coupon_amt                             AS DISCOUNT_APPLIED

FROM   SNOWFLAKE_SAMPLE_DATA.TPCDS_SF10TCL.STORE_SALES ss
JOIN   SNOWFLAKE_SAMPLE_DATA.TPCDS_SF10TCL.CUSTOMER    c
       ON ss.ss_customer_sk = c.c_customer_sk
JOIN   SNOWFLAKE_SAMPLE_DATA.TPCDS_SF10TCL.DATE_DIM     d
       ON ss.ss_sold_date_sk = d.d_date_sk
JOIN   SNOWFLAKE_SAMPLE_DATA.TPCDS_SF10TCL.ITEM         i
       ON ss.ss_item_sk = i.i_item_sk
LIMIT  5000000; 

DESC TABLE RETAIL_MARKETING_DB.CAMPAIGN.CAMPAIGN_SALES;

SELECT
    ITEM_CATEGORY,
    SALE_YEAR,
    SUM(NET_PAID)                           AS TOTAL_REVENUE,
    SUM(QUANTITY_SOLD)                      AS TOTAL_UNITS,
    AVG(SALE_PRICE)                         AS AVG_PRICE,
    COUNT(DISTINCT CUSTOMER_ID)            AS UNIQUE_CUSTOMERS,
    SUM(NET_PROFIT)                         AS TOTAL_PROFIT,
    SUM(DISCOUNT_APPLIED)                   AS TOTAL_DISCOUNTS

FROM   RETAIL_MARKETING_DB.CAMPAIGN.CAMPAIGN_SALES
WHERE  SALE_DATE BETWEEN '2000-01-01' AND '2003-12-31'
GROUP  BY ITEM_CATEGORY, SALE_YEAR
ORDER  BY TOTAL_REVENUE DESC;
